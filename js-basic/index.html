// JavaScript 基本数据类型
let num     = 3.14;           // Number
let str     = "Hello JS";     // String
let bool    = true;           // Boolean
let nothing = null;           // Null
let undef;                    // Undefined
let sym     = Symbol("id");   // Symbol（ES6）

let arr  = [1, 2, 3];         // Array（Object）
let obj  = { x: 10, y: 20 }; // Object

let types = [num, str, bool, nothing, undef, sym, arr, obj];
let names = ["num","str","bool","null","undef","sym","arr","obj"];
names.forEach((n, i) => console.log(n + " → " + typeof types[i]));

③ 表达式

// 算术表达式
let a = 10, b = 3;
console.log("a + b =", a + b);    // 13
console.log("a % b =", a % b);    // 1
console.log("a ** b =", a ** b);  // 1000（幂运算）

// 比较 & 逻辑表达式
console.log("10 == '10'  :", 10 == "10");   // true（宽松）
console.log("10 === '10' :", 10 === "10");  // false（严格）
console.log("a>5 && b<5  :", a > 5 && b < 5); // true

// 三元表达式
let grade = a >= 60 ? "及格" : "不及格";
console.log("成绩：" + a + " → " + grade);

// 模板字符串（ES6）
console.log(`a=${a}, b=${b}, a+b=${a+b}`);

④ 流程控制

// if-else
let score = 78;
if (score >= 90) {
  console.log("优秀");
} else if (score >= 70) {
  console.log("良好");   // ← 命中这里
} else if (score >= 60) {
  console.log("及格");
} else {
  console.log("不及格");
}

// switch
let day = new Date().getDay();  // 0=周日
const dayNames = ["周日","周一","周二","周三","周四","周五","周六"];
switch (day) {
  case 6:
  case 0:
    console.log(dayNames[day] + " 是周末！休息~");
    break;
  default:
    console.log(dayNames[day] + " 是工作日，加油！");
}

// for 循环
let sum = 0;
for (let i = 1; i <= 10; i++) { sum += i; }
console.log("1~10 求和 =", sum);

// while 循环
let n = 1, factorial = 1;
while (n <= 5) { factorial *= n; n++; }
console.log("5! =", factorial);

// for...of 遍历数组
const fruits = ["苹果", "香蕉", "橙子"];
for (let f of fruits) { console.log("水果：" + f); }

⑤ 函数

// 函数声明
function greet(name) {
  return "你好，" + name + "！";
}
console.log(greet("同学"));

// 函数表达式
const square = function(x) { return x * x; };
console.log("4 的平方 =", square(4));

// 箭头函数（ES6）
const add = (a, b) => a + b;
console.log("3 + 7 =", add(3, 7));

// 默认参数（ES6）
function power(base, exp = 2) { return base ** exp; }
console.log("power(3)   =", power(3));     // 9（默认 exp=2）
console.log("power(2,8) =", power(2, 8));  // 256

// 递归函数
function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}
for (let i = 0; i <= 7; i++) {
  process_stdout_write ? process_stdout_write(fibonacci(i)+" ") : null;
}
const fibArr = Array.from({length: 8}, (_, i) => fibonacci(i));
console.log("Fibonacci(0~7):", fibArr.join(", "));

// 闭包示例
function makeCounter() {
  let count = 0;
  return function() { return ++count; };
}
const counter = makeCounter();
console.log("计数器：", counter(), counter(), counter()); // 1 2 3
